У вас есть **несколько серьезных проблем** в этом коде:

## Основные проблемы:

### 1. **Асинхронная функция вызывается в синхронном контексте**

```python
async def log_meter_reading(value):  # <-- async функция
    # ... асинхронный код

def monitor_loop():
    # ... синхронный код
    await log_meter_reading(current_number)  # <-- ОШИБКА! await в синхронной функции
```

**Нельзя использовать `await` в синхронной функции!**

### 2. **Использование `async with` без асинхронного контекста**

```python
async def log_meter_reading(value):
    async with AsyncSessionLocal() as db:  # <-- требует async контекст
        async with db.begin():
            # ...
```

### 3. **Смешивание threading и asyncio**

У вас есть `threading.Thread` для `monitor_loop`, которая пытается использовать асинхронные операции.

## Решения:

### **Решение 2: Использовать asyncio правильно (сложнее)**

Если вы **обязаны** использовать асинхронность, то нужно переписать весь цикл на асинхронный:

```python
async def monitor_loop_async():
    while True:
        try:
            print("Запрос изображения с камеры...")
            # ... вся логика
            
            if is_need_add_to_history and new_digits:
                # ... другая логика
                await log_meter_reading(current_number)  # <-- await работает!

    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)
    loop.run_until_complete(monitor_loop_async())
```

### **Решение 3: Использовать `run_sync` для смешанного подхода**

```python
from meter_watch_shared.db_async import AsyncSessionLocal

async def log_meter_reading(value):
    async with AsyncSessionLocal() as db:

def _sync_log_reading(session, value):
    # Синхронный код здесь
    log = ActivityLog(
        source='meter_reader',
        event_type='meter_reading',
        data=json.dumps({'value': value}),
        meter_reading=value,
        timestamp=datetime.utcnow()

    reading = MeterReading(
        value=value,
        source='cnn_recognizer',
        timestamp=datetime.utcnow()
    )
    session.add(reading)
    session.commit()
```

```python
import time
import threading
import json
from datetime import datetime

history = []

last_image = None
last_update_value = None
history_lock = threading.Lock()
last_nearly_activity_data = None

def log_meter_reading(value):
    db = SessionLocal()
    try:

def check_history_and_save_if_needed(new_digits, img):
    """Check if digits should be saved"""
    global last_recognized_digits, history, last_nearly_activity_counter, last_image

    if len(last_recognized_digits) > 0:
        last_number = list_to_number(last_recognized_digits)
        new_number = list_to_number(new_digits)
        difference = abs(new_number - last_number)
        
        if difference > 10:
            save_test_image(img, new_number, f"big_diff_{difference}")
            if last_image is not None:
                save_test_image(last_image, new_number, f"big_diff_{difference}")

        if new_number >= last_number:
            print(f"{new_number} is greater than {last_number}")
        else:
            print(f"{new_number} is not greater than {last_number}")
            save_test_image(img, new_number, "less")            
            if last_image is not None:          
                save_test_image(last_image, new_number, "less")
        
        last_image = img            

def monitor_loop():
    """Синхронный цикл мониторинга"""
    global last_recognized_digits, history, last_nearly_activity_data, last_nearly_activity_counter, last_update_value
    
    consecutive_failures = 0
    
    while True:
        try:
            print("Запрос изображения с камеры...")
            
            img = fetch_image(config.get(ConfigKeys.CAMERA_URL) + str(timestamp_ms()))

                print("Не удалось получить изображение, пропуск цикла.")
                consecutive_failures += 1
            else:
                consecutive_failures = 0

                result, min_conf = recognize_image(img)

                new_digits = list(result['full_number'])
                current_number = list_to_number(new_digits)

                if min_conf < config.get(ConfigKeys.SAVE_THRESHOLD):
                    save_test_image(img, result['full_number'], "low_conf")

                current_time = time.time()
                time_str = time.strftime("%H:%M %d:%m:%Y", time.localtime(current_time))
                is_need_add_to_history = True

                if last_update_value is None:
                    last_update_value = {
                        "time": time_str,
                        "digits": new_digits,
                        "timestamp": current_time
                    }

                # Фильтрация дубликатов
                if len(history) >= 2:
                    last_item = list_to_number(history[-1]["digits"])

                    if current_number == last_item:
                        is_need_add_to_history = False
                    else:
                        check_history_and_save_if_needed(new_digits, img)
                        last_update_value = {
                            "time": time_str,
                            "digits": new_digits,
                            "timestamp": current_time

                            history.pop(0)
                        print("✅ Обнаружено изменение; новые цифры:", new_digits)
                        
                        # ВЫЗЫВАЕМ СИНХРОННУЮ ФУНКЦИЮ (БЕЗ await)
                        log_meter_reading(current_number)

                        save_test_image(img, result['full_number'], "next", Config.VALIDATION_DIR)

                        # RedisManager.set_key(meter_watch_shared_config.REDIS_KEYS['gas_flow'], "1")
                        # RedisManager.set_key(meter_watch_shared_config.REDIS_KEYS['gas_number'], str(result['full_number']))
                        # RedisManager.set_key(meter_watch_shared_config.REDIS_KEYS['gas_last_activity'], time_str)
                        
                        last_nearly_activity_data = {"time": time_str, "digits": new_digits}
                        last_nearly_activity_counter = 0
                    else:
                        print("⏺️ Изменений не обнаружено. Текущие цифры:", new_digits)
                        # RedisManager.set_key(meter_watch_shared_config.REDIS_KEYS['gas_flow'], "0")
                        
                        # if -1 not in new_digits:                            
                        #     last_nearly_activity_counter += 1
                        #     if last_nearly_activity_counter > 5:
                        #         last_nearly_activity_data = None

        except Exception as ex:
            print("❌ Ошибка в monitor_loop:", ex)
            consecutive_failures += 1
            
        time.sleep(Config.POLL_INTERVAL_SECONDS)

def start_monitoring():
    """Start the monitoring thread"""
    monitor_thread = threading.Thread(target=monitor_loop, daemon=True)
    monitor_thread.start()
    return monitor_thread

def get_history():
    """Get recognition history"""
    with history_lock:
        return history.copy()

def get_last_activity():
    """Get last activity data"""
    if last_nearly_activity_data and history:        
        if len(history) >= 4:
            return history[-4:], last_update_value
        else:
            return history.copy(), last_update_value
    
    return [], last_update_value
```

## Краткое резюме:

| Проблема | Решение |
|----------|---------|
| `async def log_meter_reading` | Убрать `async` → `def log_meter_reading` |
| `async with AsyncSessionLocal()` | Использовать `SessionLocal()` синхронно |
| `await log_meter_reading()` | Убрать `await` → `log_meter_reading()` |
| `AsyncSessionLocal` в импорте | Использовать `SessionLocal` из синхронного `db.py` |

**Просто уберите `async` и замените `AsyncSessionLocal` на `SessionLocal` - и всё заработает!**
