### 2. Создайте общую модель в `meter-watch-shared/models.py`:

```python
# meter-watch-shared/models.py
from sqlalchemy import Column, Integer, String, DateTime, Float, Boolean, Text
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime

Base = declarative_base()

class ActivityLog(Base):
    __tablename__ = "activity_logs"
    
    id = Column(Integer, primary_key=True)
    timestamp = Column(DateTime, default=datetime.utcnow)
    source = Column(String(50))  # 'meter' или 'person_detector'
    event_type = Column(String(50))  # 'reading', 'person_detected', 'person_left'
    data = Column(Text)  # JSON данные
    meter_reading = Column(Float, nullable=True)
    person_detected = Column(Boolean, nullable=True)

class MeterReading(Base):
    __tablename__ = "meter_readings"
    
    id = Column(Integer, primary_key=True)
    timestamp = Column(DateTime, default=datetime.utcnow)
    value = Column(Float)
    unit = Column(String(10))
    source = Column(String(50))
```

### 3. Общий модуль для работы с БД в `meter-watch-shared/db.py`:

```python

    try:
        yield db
    finally:
        db.close()
```

### 4. Создайте общий модуль для RabbitMQ в `meter-watch-shared/rabbitmq.py`:

```python
# meter-watch-shared/rabbitmq.py
import pika
import json
import os
import logging

logger = logging.getLogger(__name__)

class RabbitMQClient:
    def __init__(self):
        self.host = os.getenv('RABBITMQ_HOST', 'rabbitmq')
        self.port = int(os.getenv('RABBITMQ_PORT', 5672))
        self.user = os.getenv('RABBITMQ_USER', 'admin')
        self.password = os.getenv('RABBITMQ_PASSWORD', 'admin123')
        self.connection = None
        self.channel = None
        
    def connect(self):
        """Подключение к RabbitMQ"""
        try:
            credentials = pika.PlainCredentials(self.user, self.password)
            parameters = pika.ConnectionParameters(
                host=self.host,
                port=self.port,
                credentials=credentials,
                heartbeat=600,
                blocked_connection_timeout=300
            )
            self.connection = pika.BlockingConnection(parameters)
            self.channel = self.connection.channel()
            
            # Объявляем exchanges и queues
            self.channel.exchange_declare(
                exchange='meter_events',
                exchange_type='topic',
                durable=True
            )
            
            # Очередь для событий от счетчика
            self.channel.queue_declare(queue='meter_readings', durable=True)
            self.channel.queue_bind(
                queue='meter_readings',
                exchange='meter_events',
                routing_key='meter.reading'
            )
            
            # Очередь для событий детектора человека
            self.channel.queue_declare(queue='person_events', durable=True)
            self.channel.queue_bind(
                queue='person_events',
                exchange='meter_events',
                routing_key='person.*'
            )
            
            logger.info("Connected to RabbitMQ successfully")
            return True
        except Exception as e:
            logger.error(f"Failed to connect to RabbitMQ: {e}")
            return False
    
    def publish_event(self, routing_key, data):
        """Публикация события"""
        if not self.channel:
            if not self.connect():
                return False
        
        try:
            self.channel.basic_publish(
                exchange='meter_events',
                routing_key=routing_key,
                body=json.dumps(data),
                properties=pika.BasicProperties(
                    delivery_mode=2,  # persistent
                    content_type='application/json'
                )
            )
            logger.info(f"Published event: {routing_key}")
            return True
        except Exception as e:
            logger.error(f"Failed to publish event: {e}")
            return False
    
    def consume_events(self, queue, callback):
        """Потребление событий"""
        if not self.channel:
            if not self.connect():
                return
        
        def wrapper(ch, method, properties, body):
            try:
                data = json.loads(body)
                callback(data, method.routing_key)
                ch.basic_ack(delivery_tag=method.delivery_tag)
            except Exception as e:
                logger.error(f"Error processing message: {e}")
                ch.basic_nack(delivery_tag=method.delivery_tag, requeue=True)
        
        self.channel.basic_qos(prefetch_count=1)
        self.channel.basic_consume(
            queue=queue,
            on_message_callback=wrapper
        )
        logger.info(f"Started consuming from queue: {queue}")
        self.channel.start_consuming()
    
    def close(self):
        if self.connection:
            self.connection.close()
```

### 8. Сервис-логгер в `logger/consumer.py`:

```python
# logger/consumer.py
import sys
import json
from datetime import datetime
sys.path.append('/app/meter_watch_shared')
from rabbitmq import RabbitMQClient
from models import ActivityLog, MeterReading
from db import SessionLocal
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def process_event(data, routing_key):
    """Обработка событий и запись в БД"""
    db = SessionLocal()
    try:
        timestamp = datetime.fromisoformat(data.get('timestamp'))
        
        if routing_key == 'meter.reading':
            # Сохраняем показания счетчика
            log = ActivityLog(
                source='meter_reader',
                event_type='meter_reading',
                data=json.dumps(data),
                meter_reading=data.get('value'),
                timestamp=timestamp
            )
            db.add(log)
            
            reading = MeterReading(
                value=data.get('value'),
                unit=data.get('unit', 'kWh'),
                source=data.get('source'),
                timestamp=timestamp
            )
            db.add(reading)
            logger.info(f"Saved meter reading: {data.get('value')}")
            
        elif routing_key == 'person.detected':
            log = ActivityLog(
                source='person_detector',
                event_type='person_detected',
                data=json.dumps(data),
                person_detected=True,
                timestamp=timestamp
            )
            db.add(log)
            logger.info("Saved person detected event")
            
        elif routing_key == 'person.left':
            log = ActivityLog(
                source='person_detector',
                event_type='person_left',
                person_detected=False,
                timestamp=timestamp
            )
            db.add(log)
            logger.info("Saved person left event")
            
        db.commit()
        
    except Exception as e:
        logger.error(f"Error processing event: {e}")
        db.rollback()
    finally:
        db.close()

def main():
    mq = RabbitMQClient()
    if not mq.connect():
        logger.error("Failed to connect to RabbitMQ")
        sys.exit(1)
    
    # Потребляем все события
    mq.consume_events('meter_readings', process_event)
    mq.consume_events('person_events', process_event)

if __name__ == '__main__':
    main()
```

## Сравнение подходов:

| Характеристика | Прямое подключение | Через RabbitMQ |
|----------------|-------------------|----------------|
| **Сложность** | Просто | Средняя |
| **Надежность** | Зависит от доступности БД | Высокая (очереди) |
| **Масштабируемость** | Ограничена | Отличная |
| **Декомпозиция** | Слабая | Сильная |
| **Retry механизмы** | Нужно писать самим | Есть встроенные |
| **Разделение сервисов** | Нет | Да |

**Мой совет**: начните с **RabbitMQ**, чтобы потренироваться, он не сильно сложнее, но даст вам понимание микросервисной архитектуры и будет полезен для резюме. Для продакшена можно использовать оба подхода в зависимости от нагрузки.
